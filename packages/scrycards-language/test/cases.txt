# Name Only

Test

==>

Program(Argument)

# Name Literal Only

"Test"

==>

Program(StringLiteral)

# Tag Only

-test:test

==>

Program(Tag(Prefix,Argument,Operator,Value(String)))

# String Tag

test:"long test"

==>

Program(Tag(Argument,Operator,Value(StringLiteral)))

# Order Tag

order:cmc

==>

Program(Tag(Argument,Operator,Value(String)))

# Compound Query

t:creature or t:artifact

==>

Program(Tag(Argument, Operator, Value(String)), Or, Tag(Argument, Operator, Value(String)))

# Complex Compound Query

t:creature or (t:artifact t:vehicle)

==>

Program(Tag(Argument,Operator,Value(String)),Or,Clause(Tag(Argument,Operator,Value(String)),Tag(Argument,Operator,Value(String))))

# Complex Multi-Query

@query latest_commander_cards
-(game:mtga or game:mtgo)
-banned:commander
order:release
direction:desc
@query elves
t:elf

==>

Program(Query(QueryName,QueryBody(Clause(Prefix,Tag(Argument,Operator,Value(String)),Or,Tag(Argument,Operator,Value(String))),Tag(Prefix,Argument,Operator,Value(String)),Tag(Argument,Operator,Value(String)),Tag(Argument,Operator,Value(String)),Query(QueryName,QueryBody(Tag(Argument,Operator,Value(String)))))))
